select
                                                    name,
                                                    'category' as type 
                                            from
                                                    learndb2.category
                                            where
                                                    name ilike '%no%'
                                            union
                                            select
                                                    name,
                                                    'product' as type
                                            from
                                                    learndb2.product
                                            where
                                                    name ilike '%no%';
